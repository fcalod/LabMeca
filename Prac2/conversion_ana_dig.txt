// outside leads to ground and +5V

  pinMode(led, OUTPUT);
}

  Serial.print ("Conversion analogico-digital: ");
  Serial.print(val);
  double pot = map(val, 0, 1023, 0, 5);
  Serial.print("\nVoltaje del potenciÃ³metro ");
  Serial.print(pot);

  Serial.println();
  delay(500);
